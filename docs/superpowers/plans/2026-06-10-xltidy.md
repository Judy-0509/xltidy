# xltidy Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** xlwings로만 복잡한 조사기관 Excel을 읽어, 재사용 가능한 `TemplateSpec`을 결정론적으로 적용해 tidy long 테이블(CSV/Parquet)로 정규화하고 월/분기 버전을 통합하는 CLI + opencode/Claude skill을 만든다.

**Architecture:** `extract`(xlwings) → `CellGrid`(직렬화 pydantic) → `encode`(LLM 입력) → `TemplateSpec`(agent 또는 Qwen이 작성) → `apply`(결정론, 좌표에서 값 직접 읽기 + reconcile) → `consolidate`(버전 스택 + 드리프트). `CellGrid`를 중간 모델로 두어 `extract`를 제외한 전 로직을 Excel 없이 TDD한다. LLM은 구조/좌표만 산출하고 값은 절대 전사하지 않는다.

**Tech Stack:** Python 3.11+, xlwings, pandas, pydantic v2, pyyaml, typer, rich. extras: pyarrow(parquet), openai(qwen). dev: pytest. **금지: openpyxl, pd.read_excel/ExcelFile.**

> 참조 스펙: `docs/superpowers/specs/2026-06-10-xltidy-excel-to-db-design.md`

```python
import re

_A1 = re.compile(r"^([A-Za-z]+)(\d+)$")


def col_to_idx(letters: str) -> int:
    n = 0
    for ch in letters.upper():
        n = n * 26 + (ord(ch) - ord("A") + 1)
    return n


def idx_to_col(idx: int) -> str:
    s = ""
    while idx > 0:
        idx, rem = divmod(idx - 1, 26)
        s = chr(ord("A") + rem) + s
    return s


def a1_to_rc(a1: str) -> tuple[int, int]:
    m = _A1.match(a1.strip())
    if not m:
        raise ValueError(f"invalid A1 address: {a1!r}")
    return int(m.group(2)), col_to_idx(m.group(1))


def rc_to_a1(row: int, col: int) -> str:
    return f"{idx_to_col(col)}{row}"


def parse_range(a1: str) -> tuple[int, int, int, int]:
    if ":" in a1:
        start, end = a1.split(":", 1)
    else:
        start = end = a1
    r1, c1 = a1_to_rc(start)
    r2, c2 = a1_to_rc(end)
    return min(r1, r2), min(c1, c2), max(r1, r2), max(c1, c2)
```

```python
from __future__ import annotations

from typing import Any

from pydantic import BaseModel


class Cell(BaseModel):
    row: int  # 1-based
    col: int  # 1-based
    value: Any = None  # computed/displayed value
    formula: str | None = None
    number_format: str | None = None


class MergedRange(BaseModel):
    r1: int
    c1: int
    r2: int
    c2: int


class CellGrid(BaseModel):
    sheet: str
    n_rows: int
    n_cols: int
    cells: list[Cell] = []
    merged: list[MergedRange] = []

    def _index(self) -> dict[tuple[int, int], Any]:
        # 작은 used range 가정; 매 호출 생성으로 단순화 (필요시 캐시 도입)
        return {(c.row, c.col): c.value for c in self.cells}

    def at(self, row: int, col: int) -> Any:
        return self._index().get((row, col))

    def value_filled(self, row: int, col: int) -> Any:
        """병합셀 내부 좌표는 앵커(좌상단) 값으로 해소.

        Excel은 병합영역의 값을 앵커 셀에만 저장하므로, 세로 병합된
        카테고리 라벨 아래 행들은 at()이 None을 준다. 인덱스/라벨 읽기는
        반드시 이 메서드를 써야 조사기관 병합 표가 올바르게 풀린다.
        """
        direct = self.at(row, col)
        if direct is not None:
            return direct
        for m in self.merged:
            if m.r1 <= row <= m.r2 and m.c1 <= col <= m.c2:
                return self.at(m.r1, m.c1)
        return None
```

```python
from __future__ import annotations

from numbers import Number

from .coords import rc_to_a1
from .models import CellGrid


def _render(value) -> str:
    if value is None:
        return ""
    if isinstance(value, bool):
        return "#bool"
    if isinstance(value, Number):
        # 실제 숫자값은 노출하지 않고 형식만 (I2: LLM은 값 전사 금지)
        return "#num"
    text = str(value).strip()
    return f'"{text}"' if text else ""


def encode(grid: CellGrid) -> str:
    lines = [
        f"SHEET: {grid.sheet} (rows 1..{grid.n_rows}, cols 1..{grid.n_cols})",
    ]
    if grid.merged:
        merged = ", ".join(
            f"{rc_to_a1(m.r1, m.c1)}:{rc_to_a1(m.r2, m.c2)}" for m in grid.merged
        )
        lines.append(f"MERGED: {merged}")
    for cell in sorted(grid.cells, key=lambda c: (c.row, c.col)):
        rendered = _render(cell.value)
        if rendered == "":
            continue
        lines.append(f"{rc_to_a1(cell.row, cell.col)}\t{rendered}")
    return "\n".join(lines)
```

```python
    header: HeaderSpec
    index_columns: list[IndexColumn]
    value_block: ValueBlock
    unpivot: UnpivotSpec = UnpivotSpec()
    column_semantics: list[ColumnSemantic] = []
    period: PeriodSpec
    totals: list[TotalCheck] = []


class TemplateSpec(BaseModel):
    template_id: str
    version: int = 1
    sheet_match: SheetMatch
    tables: list[TableSpec]

    @classmethod
    def from_yaml(cls, path: str | Path) -> "TemplateSpec":
        data = yaml.safe_load(Path(path).read_text(encoding="utf-8"))
        return cls.model_validate(data)

    def to_yaml(self, path: str | Path) -> None:
        data = self.model_dump(by_alias=True, exclude_none=True)
        Path(path).write_text(
            yaml.safe_dump(data, allow_unicode=True, sort_keys=False), encoding="utf-8"
        )

    def validate_against(self, grid: CellGrid) -> list[str]:
        issues: list[str] = []
        for t in self.tables:
            r1, c1, r2, c2 = parse_range(f"{t.region.start}:{t.region.end}")
            if r2 > grid.n_rows or c2 > grid.n_cols:
                issues.append(
                    f"table {t.name}: region {t.region.start}:{t.region.end} "
                    f"out of grid bounds ({grid.n_rows}x{grid.n_cols})"
                )
        return issues
```

```python
from __future__ import annotations

import re
from dataclasses import dataclass, field

import pandas as pd

from .coords import a1_to_rc, col_to_idx, parse_range, rc_to_a1
from .models import CellGrid
from .reconcile import ReconcileReport, reconcile
from .spec import PeriodSpec, TableSpec, TemplateSpec


@dataclass
class ApplyResult:
    frame: pd.DataFrame
    reconcile: ReconcileReport
    drift: list[str] = field(default_factory=list)


def resolve_period(period: PeriodSpec, filename: str, grid: CellGrid) -> str | None:
    src = period.source
    if src.from_ == "filename" and src.pattern:
        m = re.search(src.pattern, filename or "")
        if m:
            return "-".join(g for g in m.groups() if g is not None)
    if src.cell:  # filename 실패 시 셀 폴백
        r, c = a1_to_rc(src.cell)
        val = grid.at(r, c)
        if val is not None:
            return str(val)
    return None


def _metric_name(table: TableSpec, header_row: int, col: int) -> str:
    a1 = rc_to_a1(header_row, col)
    for cs in table.column_semantics:
        if cs.source == a1:
            return cs.name
    return a1  # 폴백: 주소


def apply(grid: CellGrid, spec: TemplateSpec, *, period: str | None = None,
          filename: str = "") -> ApplyResult:
    frames: list[pd.DataFrame] = []
    for table in spec.tables:
        r1, c1, r2, c2 = parse_range(f"{table.region.start}:{table.region.end}")
        header_row = max(table.header.rows) if table.header.rows else r1
        data_start = header_row + 1
        idx_cols = [(col_to_idx(ic.col), ic.name) for ic in table.index_columns]
        vb_start, vb_end = (col_to_idx(table.value_block.cols[0]),
                            col_to_idx(table.value_block.cols[-1]))
        subtotal_labels = {t.label for t in table.totals}
        idx_name0 = table.index_columns[0].name if table.index_columns else None

        period_val = period if period is not None else resolve_period(
            table.period, filename, grid)

        records: list[dict] = []
        for row in range(data_start, r2 + 1):
            dims = {name: grid.value_filled(row, ci) for ci, name in idx_cols}  # 병합 라벨 해소
            if idx_name0 and str(dims.get(idx_name0)) in subtotal_labels:
                continue  # 소계행은 tidy 출력에서 제외 (reconcile에서만 사용)
            for vc in range(vb_start, vb_end + 1):
                value = grid.at(row, vc)
                rec = dict(dims)
                rec[table.unpivot.var_name] = _metric_name(table, header_row, vc)
                rec[table.unpivot.value_name] = value
                rec[table.period.name] = period_val
                records.append(rec)
        frames.append(pd.DataFrame.from_records(records))

    frame = pd.concat(frames, ignore_index=True) if frames else pd.DataFrame()
    rep = reconcile(grid, spec, frame)
    return ApplyResult(frame=frame, reconcile=rep)
```

```python
from __future__ import annotations

from dataclasses import dataclass, field
from numbers import Number

import pandas as pd

from .coords import col_to_idx, parse_range
from .models import CellGrid
from .spec import TableSpec, TemplateSpec


@dataclass
class ReconcileReport:
    ok: bool = True
    issues: list[str] = field(default_factory=list)


def _close(a: float, b: float) -> bool:
    return abs(a - b) <= max(1e-6, 1e-4 * abs(b))


def _check_row_subtotals(grid: CellGrid, table: TableSpec, issues: list[str]) -> None:
    r1, c1, r2, c2 = parse_range(f"{table.region.start}:{table.region.end}")
    header_row = max(table.header.rows) if table.header.rows else r1
    data_start = header_row + 1
    idx_col = col_to_idx(table.index_columns[0].col)
    vb_start = col_to_idx(table.value_block.cols[0])
    vb_end = col_to_idx(table.value_block.cols[-1])
    labels = {t.label for t in table.totals}

    subtotal_rows = [row for row in range(data_start, r2 + 1)
                     if str(grid.value_filled(row, idx_col)) in labels]
    component_rows = [row for row in range(data_start, r2 + 1)
                      if str(grid.value_filled(row, idx_col)) not in labels]

    for srow in subtotal_rows:
        label = str(grid.value_filled(srow, idx_col))
        for vc in range(vb_start, vb_end + 1):
            reported = grid.at(srow, vc)
            if not isinstance(reported, Number):
                continue
            total = sum(grid.at(row, vc) for row in component_rows
                        if isinstance(grid.at(row, vc), Number))
            if not _close(float(reported), float(total)):
                issues.append(
                    f"table {table.name} col {vc} subtotal '{label}': "
                    f"reported {reported} != sum {total}"
                )


def reconcile(grid: CellGrid, spec: TemplateSpec, frame: pd.DataFrame) -> ReconcileReport:
    issues: list[str] = []
    for table in spec.tables:
        if table.totals:
            _check_row_subtotals(grid, table, issues)
    return ReconcileReport(ok=not issues, issues=issues)
```

```python
from __future__ import annotations

from dataclasses import dataclass, field
from typing import Callable, Literal

import pandas as pd

from .apply import apply
from .coords import a1_to_rc, parse_range
from .models import CellGrid
from .spec import TemplateSpec

Extractor = Callable[..., CellGrid]


@dataclass
class ConsolidateResult:
    frame: pd.DataFrame
    drift_by_file: dict[str, list[str]] = field(default_factory=dict)


def detect_drift(grid: CellGrid, spec: TemplateSpec) -> list[str]:
    issues: list[str] = []
    for table in spec.tables:
        r1, c1, r2, c2 = parse_range(f"{table.region.start}:{table.region.end}")
        if r2 > grid.n_rows or c2 > grid.n_cols:
            issues.append(f"{table.name}: region out of bounds")
        for cs in table.column_semantics:
            if cs.source_text is None:
                continue
            r, c = a1_to_rc(cs.source)
            actual = grid.value_filled(r, c)
            if str(actual).strip() != cs.source_text.strip():
                issues.append(
                    f"{table.name}: header {cs.source} expected "
                    f"'{cs.source_text}' got '{actual}' (renamed/moved)"
                )
    return issues


def consolidate(files: list[str], spec: TemplateSpec, *,
                extractor: Extractor,
                on_drift: Literal["stop", "warn"] = "stop") -> ConsolidateResult:
    frames: list[pd.DataFrame] = []
    drift_by_file: dict[str, list[str]] = {}
    for path in files:
        grid = extractor(path)
        drift = detect_drift(grid, spec)
        if drift:
            drift_by_file[path] = drift
            if on_drift == "stop":
                continue  # 적재하지 않음
        res = apply(grid, spec, filename=path)
        frames.append(res.frame)
    frame = pd.concat(frames, ignore_index=True) if frames else pd.DataFrame()
    return ConsolidateResult(frame=frame, drift_by_file=drift_by_file)
```

```python
_INSTRUCTION = dedent(
    """
    너는 Excel 시트의 인코딩을 보고 xltidy `TemplateSpec`(JSON)을 작성한다.
    규칙:
    - 표 영역(region), 헤더 행, 행 라벨(index_columns), 값 블록(value_block),
      각 값열의 의미(column_semantics: source=헤더셀 A1, name, source_text=원본텍스트),
      unpivot(var_name/value_name), period(파일명 또는 셀), totals(소계행 라벨)을 채운다.
    - **실제 값(숫자)을 전사하지 마라.** 좌표와 구조만 출력한다(값은 인코딩에서 #num 으로 가려져 있다).
    - 출력은 오직 유효한 JSON 하나.
    파일명: {filename}

    === 시트 인코딩 ===
    {encoded}
    """
).strip()


def build_agent_prompt(encoded: str, *, filename: str = "") -> str:
    schema = json.dumps(TemplateSpec.model_json_schema(), ensure_ascii=False)
    body = _INSTRUCTION.format(filename=filename, encoded=encoded)
    return f"{body}\n\n=== TemplateSpec JSON Schema ===\n{schema}"


def infer_with_qwen(encoded: str, *, filename: str, model: str,
                    client=None, base_url: str | None = None,
                    api_key: str | None = None) -> TemplateSpec:
    if client is None:  # 지연 import: openai는 옵션 의존성
        from openai import OpenAI

        client = OpenAI(base_url=base_url, api_key=api_key or "not-needed")
    resp = client.chat.completions.create(
        model=model,
        messages=[{"role": "user", "content": build_agent_prompt(encoded, filename=filename)}],
        response_format={"type": "json_object"},
        temperature=0,
    )
    payload = json.loads(resp.choices[0].message.content)
    return TemplateSpec.model_validate(payload)
```

```python
from __future__ import annotations

import xlwings as xw

from .models import Cell, CellGrid, MergedRange


def _coerce(value):
    return value  # xlwings는 이미 계산된 표시값을 반환


def extract(path: str, sheet: str | int | None = None) -> CellGrid:
    app = xw.App(visible=False, add_book=False)
    try:
        wb = app.books.open(path, read_only=True, update_links=False)
        try:
            sht = wb.sheets.active if sheet is None else wb.sheets[sheet]
            used = sht.used_range
            n_rows = used.last_cell.row
            n_cols = used.last_cell.column
            values = used.value
            formulas = used.formula
            r0, c0 = used.row, used.column

            # used.value는 단일 셀이면 스칼라, 1행/1열이면 1D일 수 있으니 정규화
            grid_vals = _as_2d(values)
            grid_fmls = _as_2d(formulas)

            cells: list[Cell] = []
            for i, row_vals in enumerate(grid_vals):
                for j, val in enumerate(row_vals):
                    if val is None:
                        continue
                    fml = grid_fmls[i][j] if grid_fmls else None
                    cells.append(Cell(
                        row=r0 + i, col=c0 + j, value=_coerce(val),
                        formula=fml if isinstance(fml, str) and fml.startswith("=") else None,
                    ))

            merged = _read_merged(sht)
            return CellGrid(sheet=sht.name, n_rows=n_rows, n_cols=n_cols,
                            cells=cells, merged=merged)
        finally:
            wb.close()
    finally:
        app.quit()


def _as_2d(value):
    if value is None:
        return [[None]]
    if not isinstance(value, list):
        return [[value]]
    if value and not isinstance(value[0], list):
        return [value]
    return value


def _read_merged(sht) -> list[MergedRange]:
    merged: list[MergedRange] = []
    seen: set[tuple[int, int, int, int]] = set()
    used = sht.used_range
    for cell in used:  # COM 순회 (작은 used range 가정)
        area = cell.api.MergeArea
        if cell.api.MergeCells:
            r1 = area.Row
            c1 = area.Column
            r2 = r1 + area.Rows.Count - 1
            c2 = c1 + area.Columns.Count - 1
            key = (r1, c1, r2, c2)
            if key not in seen and (r1, c1) != (r2, c2):
                seen.add(key)
                merged.append(MergedRange(r1=r1, c1=c1, r2=r2, c2=c2))
    return merged
```

```python
from __future__ import annotations

import glob as _glob
from pathlib import Path

import typer
from rich import print as rprint

from .consolidate import consolidate as _consolidate
from .encode import encode as _encode
from .extract import extract as _extract  # noqa: F401 (지연 사용)
from .infer import build_agent_prompt
from .models import CellGrid
from .reconcile import reconcile  # noqa: F401
from .apply import apply as _apply
from .spec import TemplateSpec

app = typer.Typer(help="xlwings-only Excel -> tidy DB-ready tables")


def _load_grid(file: str | None, grid: str | None, sheet: str | None) -> CellGrid:
    if grid:
        return CellGrid.model_validate_json(Path(grid).read_text(encoding="utf-8"))
    if file:
        from .extract import extract
        return extract(file, sheet=sheet)
    raise typer.BadParameter("--file 또는 --grid 중 하나가 필요합니다")


def _write_frame(frame, out: str) -> None:
    if out.endswith(".parquet"):
        frame.to_parquet(out, index=False)
    else:
        frame.to_csv(out, index=False)


@app.command()
def extract(file: str, sheet: str = typer.Option(None), out: str = typer.Option(None)):
    from .extract import extract as ex
    grid = ex(file, sheet=sheet)
    data = grid.model_dump_json()
    if out:
        Path(out).write_text(data, encoding="utf-8")
    else:
        typer.echo(data)


@app.command()
def encode(file: str = typer.Option(None), grid: str = typer.Option(None),
           sheet: str = typer.Option(None)):
    g = _load_grid(file, grid, sheet)
    typer.echo(_encode(g))


@app.command()
def infer(file: str = typer.Option(None), grid: str = typer.Option(None),
          sheet: str = typer.Option(None),
          backend: str = typer.Option("agent"), out: str = typer.Option(None)):
    g = _load_grid(file, grid, sheet)
    enc = _encode(g)
    fname = file or grid or ""
    if backend == "agent":
        typer.echo(build_agent_prompt(enc, filename=fname))
        rprint("[yellow]# agent가 위 프롬프트로 spec.yaml을 작성한 뒤 "
               "`xltidy spec-validate`로 검증하세요.[/]")
    elif backend == "qwen":
        from .config import load_settings
        from .infer import infer_with_qwen
        s = load_settings()
        spec = infer_with_qwen(enc, filename=fname, model=s.model or "qwen",
                               base_url=s.base_url, api_key=s.api_key)
        text = spec.model_dump_json(indent=2)
        Path(out).write_text(text, encoding="utf-8") if out else typer.echo(text)
    else:
        raise typer.BadParameter("backend는 agent|qwen")


@app.command("spec-validate")
def spec_validate(spec: str, against: str = typer.Option(None),
                  sheet: str = typer.Option(None)):
    sp = TemplateSpec.from_yaml(spec)
    issues: list[str] = []
    if against:
        g = _load_grid(against, None, sheet)
        issues = sp.validate_against(g)
    if issues:
        for i in issues:
            rprint(f"[red]✗[/] {i}")
        raise typer.Exit(1)
    rprint("[green]✓ spec valid[/]")


@app.command()
def apply(spec: str, file: str = typer.Option(None), grid: str = typer.Option(None),
          sheet: str = typer.Option(None), period: str = typer.Option(None),
          out: str = typer.Option(None)):
    g = _load_grid(file, grid, sheet)
    sp = TemplateSpec.from_yaml(spec)
    res = _apply(g, sp, period=period, filename=(file or grid or ""))
    if not res.reconcile.ok:
        for i in res.reconcile.issues:
            rprint(f"[red]reconcile ✗[/] {i}")
    if out:
        _write_frame(res.frame, out)
        rprint(f"[green]wrote[/] {out} ({len(res.frame)} rows)")
    else:
        typer.echo(res.frame.to_csv(index=False))
    raise typer.Exit(0 if res.reconcile.ok else 2)


@app.command()
def consolidate(spec: str, files: str, out: str = typer.Option(None),
                sheet: str = typer.Option(None),
                on_drift: str = typer.Option("stop")):
    sp = TemplateSpec.from_yaml(spec)
    paths = sorted(_glob.glob(files))
    from .extract import extract as ex
    res = _consolidate(paths, sp, extractor=lambda p, sheet=sheet: ex(p, sheet=sheet),
                       on_drift=on_drift)  # type: ignore[arg-type]
    for path, drift in res.drift_by_file.items():
        rprint(f"[yellow]drift[/] {path}: {drift}")
    if out:
        _write_frame(res.frame, out)
        rprint(f"[green]wrote[/] {out} ({len(res.frame)} rows)")
    else:
        typer.echo(res.frame.to_csv(index=False))


if __name__ == "__main__":
    app()
```

```markdown
## 무엇을 하는가
xlwings로만 복잡한 Excel을 읽어 재사용 가능한 `TemplateSpec`을 적용, tidy long 테이블(CSV/Parquet)로 정규화하고 월/분기 버전을 통합한다. **LLM(나 자신)은 구조/좌표만 만들고 실제 숫자값은 절대 전사하지 않는다.**

## 워크플로 (양식당 1회 스펙 작성 → 여러 파일 적용)
1. **인코딩 확보**: `xltidy infer <대표파일.xlsx> --backend agent`
   - 출력된 시트 인코딩과 JSON Schema를 읽는다.
2. **스펙 작성(내가 직접)**: 인코딩을 해석해 `specs/<template_id>.yaml`을 작성한다.
   - region/header/index_columns/value_block/column_semantics(source_text 포함)/unpivot/period/totals 채움.
   - 숫자값은 `#num`으로 가려져 있으니 좌표·구조만 기술.
   - **다중헤더 평탄화**: 2단 헤더(예: "2024"가 C4:E4 병합 + 그 아래 "1월/2월/3월")는 값열마다
     `column_semantics` 한 항목으로 합쳐서 적는다 — `{source: "C5", name: "2024-01", source_text: "1월"}`.
     `apply`는 값열당 `column_semantics.name` 하나를 metric으로 쓰므로, 상·하위 라벨을 내가 합쳐 이름지어야 한다.
     (`header.levels`는 참고용 메타일 뿐, 평탄화는 내가 한다.)
3. **검증**: `xltidy spec-validate specs/<id>.yaml --against <대표파일.xlsx>`
4. **단일 적용 + 정합성**: `xltidy apply specs/<id>.yaml --file <한파일.xlsx> --period 2024Q1 --out out.csv`
   - reconcile ✗ 가 뜨면 스펙(소계/영역)을 고친다.
5. **버전 통합**: `xltidy consolidate specs/<id>.yaml "data/2024*.xlsx" --out merged.parquet --on-drift stop`
   - drift가 보고된 파일은 적재되지 않으니, 양식 변경이면 스펙 version을 올려 새로 작성.
```
